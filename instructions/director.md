### 品質チェックの自動化スクリプト
```bash
# SEO記事品質チェック（5分で完了）
cat > /workspace/scripts/quick-seo-check.sh << 'EOF'
#!/bin/bash
ARTICLE_FILE=$1
ARTICLE_NAME=$(basename "$ARTICLE_FILE" .md)

echo "=== SEO品質チェック: $ARTICLE_NAME ==="

# 1. 基本要件チェック
WORD_COUNT=$(wc -w < "$ARTICLE_FILE")
H2_COUNT=$(grep -c "^## " "$ARTICLE_FILE")
H3_COUNT=$(grep -c "^### " "$ARTICLE_FILE")
LINK_COUNT=$(grep -c "\[.*\](.*)" "$ARTICLE_FILE")

echo "📊 基本指標:"
echo "  文字数: $WORD_COUNT 文字"
echo "  H2見出し: $H2_COUNT 個"
echo "  H3見出し: $H3_COUNT 個"
echo "  リンク数: $LINK_COUNT 個"

# 2. SEO要素チェック
TITLE_LINE=$(head -n1 "$ARTICLE_FILE")
TITLE_LENGTH=${#TITLE_LINE}

echo "🎯 SEO要素:"
echo "  タイトル長: $TITLE_LENGTH 文字"
echo "  キーワード含有: $(grep -c "転職" "$ARTICLE_FILE") 箇所"

# 3. 品質判定
SCORE=0
[ $WORD_COUNT -ge 3000 ] && SCORE=$((SCORE+20))
[ $H2_COUNT -ge 5 ] && SCORE=$((SCORE+20))
[ $H3_COUNT -ge 10 ] && SCORE=$((SCORE+20))
[ $LINK_COUNT -ge 3 ] && SCORE=$((SCORE+20))
[ $TITLE_LENGTH -le 32 ] && SCORE=$((SCORE+20))

echo "⭐ 品質スコア: $SCORE/100"

if [ $SCORE -ge 80 ]; then
    echo "✅ 合格: 公開可能品質"
elif [ $SCORE -ge 60 ]; then
    echo "⚠️ 要修正: 軽微な改善必要"
else
    echo "❌ 不合格: 大幅修正必要"
fi
EOF
chmod +x /workspace/scripts/quick-seo-check.sh
```

### ボトルネック自動検出と対応
```bash
# 遅延検出と自動エスカレーション
cat > /workspace/scripts/bottleneck-detector.sh << 'EOF'
#!/bin/bash

# 48時間以上のタスクを検出
for writer in writer1 writer2 writer3; do
    if [ -f "./tmp/${writer}_writing.txt" ]; then
        TASK_START=$(stat -f "%Y" "./tmp/${writer}_writing.txt")
        CURRENT_TIME=$(date +%s)
        ELAPSED=$(($CURRENT_TIME - $TASK_START))
        
        # 48時間 = 172800秒
        if [ $ELAPSED -gt 172800 ]; then
            TASK_NAME=$(cat "./tmp/${writer}_writing.txt")
            
            # CMOへエスカレーション
            ./agent-send.sh cmo "【遅延アラート】

$writer のタスクが48時間超過しています。

タスク: $TASK_NAME
経過時間: $(($ELAPSED / 3600))時間
状況: 連絡途絶

対応策:
1. 他ライターへの緊急再アサイン
2. 外部ライター投入
3. タスク分割・簡素化

指示をお願いします。"

            # ライターへの確認
            ./agent-send.sh $writer "【緊急確認】

タスクが大幅に遅延しています。
現在の状況を30分以内に報告してください。

- 進捗率は？
- ブロッカーはありますか？
- サポートが必要ですか？

連絡をお待ちしています。"
        fi
    fi
done
EOF
chmod +x /workspace/scripts/bottleneck-detector.sh
```
